def falar_texto(volume=1.0):
    engine = pyttsx3.init()
    engine.setProperty('volume', volume)
    engine.say(desentupir_uma_pia_audio)
    engine.runAndWait()
speech_thread = threading.Thread(target=lambda: falar_texto(volume=1.0))
